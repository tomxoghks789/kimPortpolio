POS 시스템

개인 프로젝트로 가상의 음식점을 위해 개발하였습니다. 결제 기능은 단순하게 결제시스템과 물지않고 구현하였습니다.

이 프로젝트에 적용된 기술은 아래와 같습니다.

  • C#
  • XML
  • MS-SQL


사용자는 로그인 하여 메인 페이지에 랜딩하고 메뉴판을 추가한다던가 빈자리를 클릭하여 새로 주문을 받거나 해당 자리의 결제를 진행 할 수 있습니다.


빈자리를 클릭하면 새로운 주문을 받을 수 있으며 오른쪽에 메뉴가 뜨고 페이징 처리가 되어있습니다. 메뉴는 XML데이터로 이루어져 있습니다. 메뉴의 수량을 수정하거나 삭제한다면 XML Parser를 통해 삭제하게 됩니다.

							
// 페이징 구현
private void button32_Click(object sender, EventArgs e)
{
     if (Over_Page == true)
     {
            if (Page != Page_Index + 1)
             {
                    Page_Index++;
                    for (int u = 1; u <= 30; u++) //버튼 텍스트 초기화
                    {
                        Control control = Controls["button" + u];
                        control.Text = null;
                    }
                    XmlTextReader reader = new XmlTextReader("Menu_List.xml");
                    XmlDocument xdoc = new XmlDocument();
                    xdoc.Load(reader);
                    XmlNodeList bookNodes = xdoc.DocumentElement.SelectNodes("Menu");

                    string[] Menus = new string[30];
                    int[] Menu_Price = new int[30];

                    for (int t = 30 * Page_Index; t < (Page_Index + 1) * 30; t++)
                    {
                        for (int inloop = 0; inloop < 30; inloop++)
                        {
                            Menus[inloop] = bookNodes[t].Attributes["Name"].Value;
                        }
                    }

                    for (int d = 30 * Page_Index; d < (Page_Index + 1) * 30; d++) //가격
                    {
                        for (int inLoop = 0; inLoop < 30; inLoop++)
                        {
                            XmlNode titleNode = bookNodes[d].SelectSingleNode("Price");
                            Menu_Price[inLoop] = int.Parse(titleNode.InnerText);
                        }
                    }

                    for (int i = 1; i < 31; i++)
                    {
                        Control control = Controls["button" + i];
                        control.Text = Menus[i - 1] + "\n\n" + Menu_Price[i - 1] + "원";
                    }
                    Menu = Menus;
                    price = Menu_Price;
                }
            }
        }
